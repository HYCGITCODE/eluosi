from PyQt6.QtCore import QMimeData
from PyQt6.QtGui import QDrag

class TaskItemWidget(QWidget):
    def mouseMoveEvent(self, event):
        # 实现拖拽逻辑
        drag = QDrag(self)
        mime = QMimeData()
        mime.setData('task/id', str(self.task.id).encode())
        drag.exec(Qt.MoveAction)
