## 🎨 拖拽功能实现详情

### TaskItemWidget (task_list.py)
```python
def mouseMoveEvent(self, event):
    """启动拖拽"""
    if (event.pos() - self.drag_start_position).manhattanLength() < 10:
        return  # 避免误触
    
    drag = QDrag(self)
    mime_data = QMimeData()
    mime_data.setData('task/id', str(self.task.id).encode())
    mime_data.setData('task/date', self.task.due_date.isoformat().encode())
    drag.setMimeData(mime_data)
    
    # 创建拖拽预览图
    pixmap = QPixmap(self.size())
    self.render(pixmap)
    drag.setPixmap(pixmap)
    
    drag.exec(Qt.MoveAction)
```

### CalendarView (calendar_view.py)
```python
def dragEnterEvent(self, event):
    """拖拽进入 - 高亮显示"""
    if event.mimeData().hasFormat('task/id'):
        event.acceptProposedAction()
        self.setCursor(Qt.OpenHandCursor)

def dropEvent(self, event):
    """拖拽放置 - 更新任务日期"""
    task_id = int(event.mimeData().data('task/id').data())
    
    # 获取放置位置的日期
    pos = self.calendar_table.viewport().mapFromGlobal(event.globalPos())
    item = self.calendar_table.itemAt(pos)
    
    if item and item.text().isdigit():
        day = int(item.text())
        new_date = QDate(self.year, self.month, day).toPyDate()
        
        # 回调主窗口更新数据库
        if hasattr(self, 'on_task_dropped'):
            self.on_task_dropped(task_id, new_date)
        
        QToolTip.showText(event.globalPos(), f"任务已移动到 {new_date}")
```

### MainWindow (main_window.py)
```python
def _on_task_dropped(self, task_id: int, new_date):
    """处理拖拽放置"""
    self.task_service.update_task(task_id, due_date=new_date)
    self._refresh_task_list()
    self.status_bar.showMessage(f"✓ 任务已移动到 {new_date}", 3000)
```
